Struktur Kolom Sheet

tanggal, nama, nisn, kelas, mapel, bab, skor, status

Pasang Apps Script

Di Spreadsheet Anda, klik menu Ekstensi > Apps Script. Hapus semua kode bawaan lalu paste kode di bawah ini:

function doGet(e) {
  var action = e.parameter.action;
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheetSoal = ss.getSheetByName("DataSoal");
  var sheetSkor = ss.getSheetByName("Skor");
  
  if (action == "getMetadata") {
    var data = sheetSoal.getDataRange().getValues();
    var header = data[0];
    var listKelas = [];
    var listMapel = [];
    var listBab = [];
    
    var idxKelas = header.indexOf("kelas");
    var idxMapel = header.indexOf("mapel");
    var idxBab = header.indexOf("bab");
    
    for (var i = 1; i < data.length; i++) {
      var k = data[i][idxKelas];
      var m = data[i][idxMapel];
      var b = data[i][idxBab];
      if (k && listKelas.indexOf(k) === -1) listKelas.push(k);
      if (m && listMapel.indexOf(m) === -1) listMapel.push(m);
      if (b && listBab.indexOf(b) === -1) listBab.push(b);
    }
    
    // Kirim opsi-opsi unik
    var result = {
      kelas: listKelas,
      mapel: listMapel,
      bab: listBab,
      relations: getRelations(data, idxKelas, idxMapel, idxBab)
    };
    
    return ContentService.createTextOutput(JSON.stringify({status: "success", data: result}))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  if (action == "getQuestions") {
    var filterKelas = e.parameter.kelas;
    var filterBab = e.parameter.bab;
    var filterMapel = e.parameter.mapel;
    
    var data = sheetSoal.getDataRange().getValues();
    var header = data[0];
    
    var idxMapel = header.indexOf("mapel");
    var idxKelas = header.indexOf("kelas");
    var idxBab = header.indexOf("bab");
    var idxSoal = header.indexOf("soal");
    var idxGambar = header.indexOf("gambar");
    var idxA = header.indexOf("opsi_a");
    var idxB = header.indexOf("opsi_b");
    var idxC = header.indexOf("opsi_c");
    var idxD = header.indexOf("opsi_d");
    var idxJawaban = header.indexOf("jawaban");
    
    var filtered = [];
    for (var i = 1; i < data.length; i++) {
      var r = data[i];
      if (String(r[idxKelas]).trim() === String(filterKelas).trim() && 
          String(r[idxBab]).trim() === String(filterBab).trim()) {
        filtered.push({
          id: i, // Nomor unik baris sebagai ID
          mapel: r[idxMapel],
          kelas: r[idxKelas],
          bab: r[idxBab],
          soal: r[idxSoal],
          gambar: r[idxGambar] || "",
          opsi_a: r[idxA],
          opsi_b: r[idxB],
          opsi_c: r[idxC],
          opsi_d: r[idxD],
          jawaban: r[idxJawaban]
        });
      }
    }
    
    return ContentService.createTextOutput(JSON.stringify({status: "success", data: filtered}))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  if (action == "getScores") {
    var nisn = e.parameter.nisn;
    var data = sheetSkor.getDataRange().getValues();
    var header = data[0];
    var filtered = [];
    
    var idxTime = header.indexOf("tanggal");
    var idxNama = header.indexOf("nama");
    var idxNisn = header.indexOf("nisn");
    var idxKelas = header.indexOf("kelas");
    var idxMapel = header.indexOf("mapel");
    var idxBab = header.indexOf("bab");
    var idxSkor = header.indexOf("skor");
    var idxStatus = header.indexOf("status");
    
    for (var i = 1; i < data.length; i++) {
      var r = data[i];
      if (String(r[idxNisn]).trim() === String(nisn).trim()) {
        filtered.push({
          timestamp: Utilities.formatDate(new Date(r[idxTime]), "GMT+7", "dd/MM/yyyy HH:mm"),
          nama: r[idxNama],
          nisn: r[idxNisn],
          kelas: r[idxKelas],
          mapel: r[idxMapel],
          bab: r[idxBab],
          skor: r[idxSkor],
          status: r[idxStatus] || (r[idxSkor] >= 80 ? "Lulus" : "Remedial")
        });
      }
    }
    
    return ContentService.createTextOutput(JSON.stringify({status: "success", data: filtered}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheetSkor = ss.getSheetByName("Skor");
  var postData = JSON.parse(e.postData.contents);
  
  if (postData.action == "submitScore") {
    sheetSkor.appendRow([
      new Date(),
      postData.nama,
      postData.nisn,
      postData.kelas,
      postData.mapel,
      postData.bab,
      postData.skor,
      postData.status
    ]);
    return ContentService.createTextOutput(JSON.stringify({status: "success"}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function getRelations(data, idxKelas, idxMapel, idxBab) {
  var rel = {};
  for (var i = 1; i < data.length; i++) {
    var k = data[i][idxKelas];
    var m = data[i][idxMapel];
    var b = data[i][idxBab];
    if (k && m && b) {
      if (!rel[k]) rel[k] = {};
      if (!rel[k][m]) rel[k][m] = [];
      if (rel[k][m].indexOf(b) === -1) rel[k][m].push(b);
    }
  }
  return rel;
}

  • Jalankan sebagai: Saya (Email anda)
